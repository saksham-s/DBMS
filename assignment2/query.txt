CREATE DATABASE dbms;
USE dbms;
CREATE TABLE drivers(
   did    INTEGER  NOT NULL PRIMARY KEY 
  ,dname  VARCHAR(30)
  ,rating INTEGER 
  ,age    INTEGER 
);

CREATE TABLE reserves(
   did     INTEGER  NOT NULL
  ,cid     INTEGER  NOT NULL
  ,charges INTEGER
  ,PRIMARY KEY(did,cid)
);

CREATE TABLE cars(
   cid    INTEGER  NOT NULL PRIMARY KEY 
  ,cname  VARCHAR(25)
  ,colour VARCHAR(10)
);
